虾咩是Pygame？

开始变得好玩了一点...

我们开始慢慢的丰富这个程序，现在已经掌握了一些游戏制作的技巧了：

  • 图片的缩放和绘制位置
  • 鼠标图标设置并跟随鼠标移动
  • 文本的显示与效果
  • 按键绑定和事件触发

import pygame, sys
#导入pygame库
from pygame.locals import *
#导入一些pygame常用的函数和常量

# 设置窗口大小
WINDOWWIDTH = 1024
WINDOWHEIGHT = 600

# new: 文本颜色的RGB值：黄色
TEXTCOLOR = (255, 255, 0)
# 红色的RGB值 RED = (255, 0, 0)

# new: 文字对象的文本内容、字体、颜色和位置的设置
# new: 以文字对象左上角的位置为原点进行绘制
def drawText(text, font, surface, x, y):
    textobj = font.render(text, 1, TEXTCOLOR)
    textrect = textobj.get_rect()
    textrect.topleft = (x, y)
    surface.blit(textobj, textrect)

# new: 游戏终止函数
def gameOver():
    pygame.quit()
    sys.exit()


#初始化pygame,为使用硬件做准备
pygame.init()

# 创建了一个Surface窗口对象，大小为设置的宽度和高度
# 这个Surface对象会经常用到，我们可以在上面进行绘制
windowSurface = pygame.display.set_mode((WINDOWWIDTH, WINDOWHEIGHT))#, pygame.FULLSCREEN)

# 设置了窗口标题
pygame.display.set_caption('豌豆射手大战妖艳贱货')

# new:设置字体和大小
font = pygame.font.SysFont('SimHei', 32)

# 加载背景图片，获取图片对象
backgroundImage = pygame.image.load('resources\\background.png')
# new: 使用pygame.transform.scale缩放原来的图片，并获取新的图片对象
rescaledBackground = pygame.transform.scale(backgroundImage, (WINDOWWIDTH, WINDOWHEIGHT))

# new: 加载豌豆射手图片 并获取图片的边距
playerImage = pygame.image.load('resources\\SnowPea.png')
playerRect = playerImage.get_rect()

# new: 加载妖艳贱货图片 并获取图片的边距
zombieImage = pygame.image.load('resources\\NormalZombie.png')
zombieRect = zombieImage.get_rect()

# new: 加载鼠标图片，获取图片对象
mouseImage = pygame.image.load('resources\\mouse.png')

#游戏主循环
while True:
    # new: 玩家按键判断（后面将单独做个函数处理）
    # new: 如果关闭窗口或按下键盘Esc键，则退出游戏，
    # new: 如果按下键盘回车键盘，则执行后面的语句
    for event in pygame.event.get():
            if event.type == QUIT:
                gameOver()
            if event.type == KEYDOWN:
                if event.key == K_ESCAPE: 
                    gameOver()
                if event.key == K_RETURN:
                    pass

    # 将背景图片对象绘制到窗口上，并全窗口覆盖
    windowSurface.blit(rescaledBackground, (0, 0))

    # new: 将豌豆射手绘制到窗口上，位置为： 窗口中间 和 下方 -70像素
    windowSurface.blit(playerImage, (WINDOWWIDTH / 3.5, WINDOWHEIGHT - 70))

    # new: 将妖艳贱货绘制到窗口上，位置为： x轴 1024/1.5 像素， y轴 -90像素
    windowSurface.blit(zombieImage, (WINDOWWIDTH / 1.5, WINDOWHEIGHT - 90))

    # new: 调用drawText()方法绘制文字到屏幕上
    drawText('豌豆射手大战妖艳贱货', font, windowSurface, (WINDOWWIDTH / 4), (WINDOWHEIGHT / 4))
    drawText('按下回车开始游戏（按Esc键退出游戏）', font, windowSurface, (WINDOWWIDTH / 3) - 10, (WINDOWHEIGHT / 3) + 50)

    # new: 获得鼠标x轴和y轴位置
    x, y = pygame.mouse.get_pos()

    # new: 获取鼠标图片左上角坐标，从原点到左上角
    # new: get_width()和get_height() 是获取图片的宽和高 
    x -= mouseImage.get_width() / 2
    y -= mouseImage.get_height() / 2

    # new: 把鼠标图片画上去
    windowSurface.blit(mouseImage, (x, y))

    # 刷新一下屏幕，重新绘制画面
    pygame.display.update()
